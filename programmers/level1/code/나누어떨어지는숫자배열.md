### c++
```
#include <string>
#include <vector>
#include <algorithm>
 
using namespace std;
 
vector<int> solution(vector<int> arr, int divisor) {
    vector<int> answer;
    
    for(int i=0; i<arr.size(); i++) {
        if(arr[i] % divisor == 0) {
            answer.push_back(arr[i]);
        }
    }
    
    if(answer.size() == 0) {
        answer.push_back(-1);
    }
    else {
        sort(answer.begin(), answer.end());
    }
    
    return answer;
}
```

### java
```
import java.util.*;
 
class Solution {
    public int[] solution(int[] arr, int divisor) {
        ArrayList<Integer> tmp = new ArrayList<Integer>();
        
        for(int i=0; i<arr.length; i++) {
            if(arr[i] % divisor == 0) {
                tmp.add(arr[i]);
            }
        }
        
        int[] answer;
        
        if(tmp.size() == 0) {
            answer = new int[1];
            answer[0] = -1;
        }
        else {
            answer = new int[tmp.size()];
            for(int i=0; i<tmp.size(); i++) {
                answer[i] = tmp.get(i);
            }
            Arrays.sort(answer);
        }        
        
        return answer;
    }
}
```

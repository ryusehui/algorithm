## 코드
### c++
```c++
#include <algorithm>
#include <vector>
 
using namespace std;
 
string solution(vector<string> seoul) {
    string answer = "김서방은 ";

    for(int i=0; i<seoul.size(); i++) {
        if(seoul[i] == "Kim")
            answer += to_string(i);
    }
    
    return answer + "에 있다";
}
```

### java
```java
class Solution {
  public String solution(String[] seoul) {
      String answer = "김서방은 ";
      
      for(int i=0; i<seoul.length; i++) {
          if(seoul[i].equals("Kim"))
              answer += Integer.toString(i);
      }
      
      return answer + "에 있다";
  }
}
```

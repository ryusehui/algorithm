## 코드
### c++
```c++
using namespace std;
 
int sum = 0;
int getSum(int n) {
    if(n < 10)
        return sum + n;
    sum += n%10;
    return getSum(n/10);
}
 
bool solution(int x) {
    if(x % getSum(x) == 0)
        return true;
    return false;
}
```

### java
```java
class Solution {
    int sum = 0;
    int getSum(int n) {
        if(n < 10)
            return sum + n;
        sum += n%10;
        return getSum(n/10);
    }
    
    public boolean solution(int n) {
        if(n % getSum(n) == 0)
            return true;
        return false;
    }
}
```

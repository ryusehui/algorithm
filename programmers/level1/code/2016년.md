### javascript
```
function solution(a, b) {
    var answer = '';
    
    let days = ['MON', 'TUE', 'WED','THU', 'FRI', 'SAT', 'SUN'];
    let datesPerMonth = [31,29,31,30,31,30,31,31,30,31,30,31];
    
    let date = b;
    for(let i=0; i<a-1; i++) {
        date += datesPerMonth[i];
    }
    
    answer = days[(date+3) % days.length]      
    
    return answer;
}
```

## 코드
### c++
```c++
#include <iostream>
 
using namespace std;
 
int main(void) {
    int a;
    int b;
    cin >> a >> b;
    
    for(int i=0; i<b; i++) {
        for(int j=0; j<a; j++)
            cout << "*";
        cout << "\n";
    }
    
    return 0;
}
```

### java
```java
import java.util.Scanner;
 
public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
 
        for(int i=0; i<b; i++) {
            for(int j=0; j<a; j++)
                System.out.print("*");
            System.out.println("");
        }
    }
}
```

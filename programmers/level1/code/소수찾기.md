## 코드
### c++
```c++
#include <string>
#include <cmath>
 
using namespace std;
 
int solution(int n) {
    int answer =0 ;
    
    int check[1000001]={0,};
    for(int i=2; i<=n; i++){
        if(!check[i])answer++;
        for(int j=1; i*j<=n;j++)check[i*j]=1;
    }
    
    return answer;
}
```

### java
```java
class Solution {
  public int solution(int n) {
        int[] prime = new int[n+1];
        int i, j, cnt = 0;
        
        for(i=0; i<=n; i++) prime[i] = 0;
        prime[1] = 1;
        
        for(i=2; i<=n; i++) {
            for(j=2; i*j<=n; j++)
                prime[i*j] = 1;
        }
        
        for(i=1; i<=n; i++)
            if(prime[i] != 1) cnt++;
        
        return cnt;
    }
}
```

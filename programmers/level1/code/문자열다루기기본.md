### c++
```
#include <string>
 
using namespace std;
 
bool solution(string s) {
    if(s.size() == 4 || s.size() == 6) {
        for(int i=0; i<s.size(); i++) {
            if(s[i] < '0' || '9' < s[i])
                return false;
        }
        return true;
    }
    return false;
}
```

### java
```
class Solution {
  public boolean solution(String s) {
      
      if(s.length() == 4 || s.length() == 6) {
          for(int i=0; i<s.length(); i++) {
              if(!Character.isDigit(s.charAt(i)))
                  return false;
          }
          return true;
      }
      return false;
  }
}
```

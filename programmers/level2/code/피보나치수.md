### c++
```
using namespace std;
 
int solution(int n) {
    int answer = 0;
    int MOD = 1234567;
    int num0 = 0;
    int num1 = 1;
    
    if(n == 0)
        return 0;
    else if(n ==1 )
        return 1;
    
    for(int i=2; i<=n; i++) {
        answer = num0 % MOD + num1 % MOD;
        num0 = num1;
        num1 = answer;
    }
    
    return answer % MOD;
}
```

### java
```java
class Solution {
    public int solution(int n) {
        int MOD = 1234567;
        if (n == 0)
            return 0;
        if (n == 2 || n == 1)
            return 1;
 
        int[] arr = new int[n];
        arr[0] = 1;
        arr[1] = 1;
 
        for (int i = 2; i < arr.length; i++)
            arr[i] = arr[i - 1] % MOD + arr[i - 2] % MOD;
 
        return arr[n - 1] % MOD;
    }
}
```
